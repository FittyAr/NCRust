use crate::fs::archive::{extract_archive, compress_zip};

pub fn spawn_extract_task(
    archive_path: PathBuf,
    destination_dir: PathBuf,
) -> mpsc::Receiver<ProgressUpdate> {
    let (tx, rx) = mpsc::channel(100);
    tokio::spawn(async move {
        if let Err(e) = extract_archive(&archive_path, &destination_dir, &tx).await {
            let _ = tx.send(ProgressUpdate {
                current_file: archive_path.to_string_lossy().into_owned(),
                files_copied: 0,
                total_files: 0,
                bytes_copied: 0,
                total_bytes: 0,
                error: Some(format!("Extraction failed: {}", e)),
            }).await;
        } else {
             let _ = tx.send(ProgressUpdate {
                current_file: "Completed".to_string(),
                files_copied: 1,
                total_files: 1,
                bytes_copied: 0,
                total_bytes: 0,
                error: None,
            }).await;
        }
    });
    rx
}

pub fn spawn_compress_task(
    sources: Vec<PathBuf>,
    dest_archive: PathBuf,
) -> mpsc::Receiver<ProgressUpdate> {
    let (tx, rx) = mpsc::channel(100);
    tokio::spawn(async move {
        if let Err(e) = compress_zip(sources, &dest_archive, &tx).await {
             let _ = tx.send(ProgressUpdate {
                current_file: dest_archive.to_string_lossy().into_owned(),
                files_copied: 0,
                total_files: 0,
                bytes_copied: 0,
                total_bytes: 0,
                error: Some(format!("Compression failed: {}", e)),
            }).await;
        } else {
             let _ = tx.send(ProgressUpdate {
                current_file: "Completed".to_string(),
                files_copied: 1,
                total_files: 1,
                bytes_copied: 0,
                total_bytes: 0,
                error: None,
            }).await;
        }
    });
    rx
}
